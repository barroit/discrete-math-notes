\subsection{要素の個数}

\begin{topic}[有限集合]
   要素の数が有限
   \[
      \begin{aligned}
         A_1 &= \set{2, 4, 6, \omitelement, 100} \\
         B_1 &= \set{1, 3, 5, \omitelement, 99}
      \end{aligned}
   \]

   要素の数を $|A|$ で表す
\end{topic}

\begin{topic}[無限集合]
   要素の数が無限
   \[
      \begin{aligned}
         A_2 &= \set{2, 4, 6, \omitelement} \\
         B_2 &= \set[m = 2k - 1,\;k \in \mathbb{N}]{m}
      \end{aligned}
   \]

   $\mathbb{N}, \mathbb{Z}, \mathbb{Q}, \mathbb{R}, \mathbb{C}$はすべて無限集合
\end{topic}

\begin{theorem}
   \leavevmode

   \begin{itemtopic}
      \item $U$ を有限な全体集合とする

      \item $A, B$ を $U$ の部分集合とするとき
      \begin{itemtopic}
         \item $|A \cup B| = |A| + |B| - |A \cap B|$
         \item $|\overline{A}| = |U| - |A|$
      \end{itemtopic}
   \end{itemtopic}
\end{theorem}
